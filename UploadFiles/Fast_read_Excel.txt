Private Function fastReadAsXlsx(ByVal excelPath As String, ByVal sheetNumber As Integer) As Object
        Dim _excel As New Microsoft.Office.Interop.Excel.Application

        Dim firstcell As Microsoft.Office.Interop.Excel.Range
        Dim lastcell As Microsoft.Office.Interop.Excel.Range
        Dim myArray(,) As Object
        Dim wBook As Microsoft.Office.Interop.Excel.Workbook = _excel.Workbooks.Open(excelPath)
        Dim wSheet As Microsoft.Office.Interop.Excel.Worksheet = wBook.Sheets(sheetNumber)
        Dim eRange As Microsoft.Office.Interop.Excel.Range

        firstcell = wSheet.Cells(1, 1)
        lastcell = wSheet.Cells(wSheet.UsedRange.Rows.Count + 1, wSheet.UsedRange.Columns.Count + 1)
        eRange = wSheet.Range(firstcell, lastcell)
        myArray = eRange.Value

        wBook.Close()
        _excel.Quit()




        Return myArray
    End Function
